1. 不改函数签名和模块级变量名

3. 加段落注释：`// ── 状态 ──` `// ── 回调 ──` `// ── 渲染 ──`
4. 保持 vendor/React 用法不变：不动 Y.useState、X.jsx 等
5. 不改 JSX 结构

### 重写代码

| urlToImageNode | UrlToImageNodeComp |
```js
  // ── 数据读取 ──
  let {
      updateNodeData
    } = Gt(),
    sources = ut(t({
      handleType: `target`
    }).map(e => e.source)),
    localEngine = Uc(),
    // ── 状态：输入 URL、加载标志、错误信息 ──
    [inputUrl, setInputUrl] = Y.useState(n.inputUrl || ``),
    [loading, setLoading] = Y.useState(false),
    [errorMsg, setErrorMsg] = Y.useState(null);
  // ── 副作用：从上游文本节点提取 http 开头的 URL ──
  Y.useEffect(() => {
    let urlText = ``;
    for (let src of sources) if (src?.data?.text && typeof src.data.text == `string` && src.data.text.startsWith(`http`)) {
      urlText = src.data.text;
      break;
    }
    urlText && urlText !== inputUrl && (setInputUrl(urlText), updateNodeData(e, {
      inputUrl: urlText
    }));
  }, [sources, e, updateNodeData, inputUrl]);
  // ── 回调：拉取图片 URL 并转 base64 ──
  let fetchImage = async () => {
    if (inputUrl) {
      setLoading(true), setErrorMsg(null);
      try {
        let response = await zc(inputUrl, {
          method: `GET`,
          localPort: localEngine.status.isConnected ? localEngine.status.port : undefined
        });
        if (!response.ok) throw Error(`HTTP ${response.status}`);
        let imageBlob = await response.blob();
        updateNodeData(e, {
          imageUrl: await new Promise((resolve, reject) => {
            let reader = new FileReader();
            reader.onload = () => resolve(reader.result), reader.onerror = reject, reader.readAsDataURL(imageBlob);
          })
        }), n.onShowToast?.(`图片转换成功`);
      } catch (err) {
        console.error(err), setErrorMsg(err.message || `转换失败`), updateNodeData(e, {
          imageUrl: null
        });
      } finally {
        setLoading(false);
      }
    }
  };
  // ── 副作用：inputUrl 变化时自动拉取（去重） ──
  return Y.useEffect(() => {
    inputUrl && inputUrl !== n.lastFetchedUrl && fetchImage().then(() => {
      updateNodeData(e, {
        lastFetchedUrl: inputUrl
      });
    });
  }, [inputUrl, n.lastFetchedUrl]),
  // ── 渲染 ──
  X.jsxs(`div`, {
    className: `relative flex flex-col`,
    children: [X.jsx(si, {
      id: e,
      data: n,
      defaultTitle: `网址转图片`,
      icon: X.jsx(kn, {
        size: 11,
        className: `text-gray-500`
      })
    }), X.jsxs(`div`, {
      className: `w-[260px] bg-[#1a1a1a] rounded-xl shadow-2xl border-2 transition-colors ${r ? `border-[#555]` : `border-[#333] hover:border-[#444]`}`,
      children: [X.jsx(_r, {
        type: `target`,
        position: J.Left
      }), X.jsxs(`div`, {
        className: `p-3 space-y-3`,
        children: [X.jsxs(`div`, {
          className: `flex gap-2`,
          children: [X.jsx(`input`, {
            type: `text`,
            value: inputUrl,
            onChange: t => {
              setInputUrl(t.target.value), updateNodeData(e, {
                inputUrl: t.target.value
              });
            },
            className: `flex-1 bg-[#111] border border-[#333] rounded px-2 py-1 text-xs text-gray-300 focus:outline-none focus:border-gray-500 nodrag`,
            placeholder: `输入图片 URL (或连线传入)`
          }), X.jsx(`button`, {
            onClick: fetchImage,
            disabled: loading || !inputUrl,
            className: `px-2 py-1 bg-[#333] hover:bg-[#444] rounded text-gray-300 disabled:opacity-50 transition-colors`,
            title: `重新获取`,
            children: X.jsx(L, {
              size: 14,
              className: loading ? `animate-spin` : ``
            })
          })]
        }), errorMsg && X.jsxs(`div`, {
          className: `flex items-center gap-1 text-red-400 text-[10px]`,
          children: [X.jsx(pt, {
            size: 12
          }), X.jsx(`span`, {
            children: errorMsg
          })]
        }), X.jsx(`div`, {
          className: `border border-[#333] rounded-lg overflow-hidden bg-[#111] relative aspect-video flex items-center justify-center`,
          children: loading ? X.jsxs(`div`, {
            className: `text-xs text-gray-500 flex items-center gap-2`,
            children: [X.jsx(L, {
              size: 14,
              className: `animate-spin`
            }), `转换中...`]
          }) : n.imageUrl ? X.jsx(`img`, {
            src: n.imageUrl,
            loading: `lazy`,
            decoding: `async`,
            className: `w-full h-full object-contain cursor-pointer`,
            onDoubleClick: e => {
              e.stopPropagation(), n.onZoom && n.onZoom(n.imageUrl);
            }
          }) : X.jsxs(`div`, {
            className: `text-[10px] text-gray-600 flex flex-col items-center gap-1`,
            children: [X.jsx(Ot, {
              size: 20,
              className: `opacity-50`
            }), X.jsx(`span`, {
              children: `等待图片 URL`
            })]
          })
        })]
      }), X.jsx(_r, {
        type: `source`,
        position: J.Right,
```
